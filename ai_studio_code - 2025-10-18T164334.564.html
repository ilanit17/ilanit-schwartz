מָה זֶה חֵקֶר?

חֵקֶר הוּא מַסָּע מְרַתֵּק שֶׁל לְמִידָה!

בְּחֵקֶר, אַתֶּם הוֹפְכִים לְחוֹקְרִים אֲמִתִּיִּים! 🔬

אַתֶּם בּוֹחֲרִים נוֹשֵׂא שֶׁמְּעַנְיֵן אֶתְכֶם, שׁוֹאֲלִים שְׁאֵלוֹת, מְחַפְּשִׂים מֵידָע וּמְגַלִּים דְּבָרִים חֲדָשִׁים!

💡 לָמָּה חֵקֶר זֶה כֵּיף?
  • לוֹמְדִים עַל מַה שֶּׁבֶּאֱמֶת מְעַנְיֵן אוֹתָנוּ.
  • הוֹפְכִים לְמוּמְחִים בְּנוֹשֵׂא שֶׁבָּחַרְנוּ.
  • מְגַלִּים דְּבָרִים חֲדָשִׁים וּמַפְתִּיעִים.
  • יוֹצְרִים תּוֹצָר מְיֻחָד מִשֶּׁלָּנוּ.

הַכָּרַת עוֹלַם הַחַי - דַּפֵּי עֲבוֹדָה

🐾 דַּף עֲבוֹדָה 1: מָה זֶה בַּעַל חַיִּים?

📖 קְטַע מֵידָע

בַּעַל חַיִּים הוּא יְצוּר חַי, וְגַם הַצֶּמַח הוּא יְצוּר חַי. שְׁנֵיהֶם זְקוּקִים לְאֶנֶרְגְּיָה כְּדֵי לְהִתְקַיֵּם, אַךְ כָּל אֶחָד מֵהֶם שׁוֹאֵב אוֹתָהּ מִמָּקוֹר אַחֵר. צְמָחִים מְנַצְּלִים אֶת אֶנֶרְגְּיַת הַשֶּׁמֶשׁ, אַךְ הַחַיּוֹת מְקַבְּלוֹת אֶת הָאֶנֶרְגְּיָה הַדְּרוּשָׁה לָהֶן מֵהַמָּזוֹן שֶׁהֵן אוֹכְלוֹת. חֵלֶק מֵהַחַיּוֹת נִזּוֹנוֹת מִצְּמָחִים, חֵלֶק מִבַּעֲלֵי חַיִּים אֲחֵרִים, וְיֵשׁ חַיּוֹת הָאוֹכְלוֹת גַּם צְמָחִים וְגַם בָּשָׂר. יֵשׁ רַבִּים מְאֹד מִינֵי בַּעֲלֵי הַחַיִּים, וְהַמַּדְעָנִים מְמַיְּנִים אוֹתָם לִסְדָרוֹת, לְמִשְׁפָּחוֹת, לְסוּגִים וּלְמִינִים.

💡 פִּנַּת אוֹצַר מִילִּים

אֶנֶרְגְּיָה: כֹּחַ, מֶרֶץ.

📝 הָשִׁיבוּ עַל הַשְּׁאֵלוֹת בְּמַחְבֶּרֶת שָׂפָה

🤔 שְׁאֵלוֹת לְמַחְשָׁבָה

  1. עַל אֵיזוֹ שְׁאֵלָה עוֹנֶה קְטַע הַמֵּידָע?
  2. לְמָה זְקוּקִים גַּם הַצֶּמַח וְגַם בַּעֲלֵי הַחַיִּים כְּדֵי לִחְיוֹת?
  3. מֵאַיִן מְקַבֵּל הַצֶּמַח אֶת הָאֶנֶרְגְּיָה שֶׁלּוֹ?
  4. מֵאַיִן מְקַבְּלוֹת הַחַיּוֹת אֶת הָאֶנֶרְגְּיָה שֶׁלָּהֶן?
  5. לְפִי הַקֶּטַע, מִמָּה נִזּוֹנוֹת הַחַיּוֹת? (יֵשְׁנָן 3 אֶפְשָׁרוּיוֹת).
  6. בַּקֶּטַע כָּתוּב: "הַמַּדְעָנִים מְמַיְּנִים אֶת מִינֵי בַּעֲלֵי הַחַיִּים". הַצִּיעוּ דֶּרֶךְ מִשֶּׁלָּכֶם לְמַיֵּן 10 בַּעֲלֵי חַיִּים שׁוֹנִים בְּטַבְלָה.

🦁 דַּף עֲבוֹדָה 2: מַהוּ יוֹנֵק?

📖 קְטַע מֵידָע

יוֹנֵק הוּא בַּעַל חַיִּים בַּעַל שֵׂעָר אוֹ פַּרְוָה, הָעוֹזְרִים לוֹ לִשְׁמֹר עַל חֹם גּוּפוֹ. וַלְדוֹתֵיהֶם (הַתִּינוֹקוֹת) שֶׁל הַיּוֹנְקִים נִזּוֹנִים מֵחָלָב הַנּוֹצָר בְּגוּף אִמָּם. הַיּוֹנְקִים חַיִּים בְּכָל מָקוֹם עַל פְּנֵי כַּדּוּר הָאָרֶץ, גַּם בְּאַנְטְאַרְקְטִיקָה הַקְּפוּאָה וְגַם בְּאַפְרִיקָה הַחַמָּה. מַרְבִּיתָם שׁוֹכְנִים בַּיַּבָּשָׁה. יֵשׁ יוֹנְקִים מְסֻיָּמִים, כְּגוֹן לִוְיָתָן, הַחַיִּים בַּמַּיִם. עֲטַלֵּף הוּא הַיּוֹנֵק הַיָּחִיד הַמְּסֻגָּל לָעוּף.

💡 פִּנַּת אוֹצַר מִילִּים

וַלְדוֹתֵיהֶם: הַתִּינוֹקוֹת שֶׁלָּהֶם.

📝 הָשִׁיבוּ עַל הַשְּׁאֵלוֹת בְּמַחְבֶּרֶт שָׂפָה

🤔 שְׁאֵלוֹת לְמַחְשָׁבָה

  1. מַהוּ יוֹנֵק? הַסְבִּירוּ בְּמִילִּים שֶׁלָּכֶם.
  2. לְמָה מְשַׁמְּשִׁים הַשֵּׂעָר אוֹ הַפַּרְוָה שֶׁעַל גּוּף הַיּוֹנֵק?
  3. מִמָּה נִזּוֹנִים הַתִּינוֹקוֹת שֶׁל הַיּוֹנְקִים?
  4. הֵיכָן חַיִּים הַיּוֹנְקִים? תְּנוּ דֻּגְמָה לְמָקוֹם חַם וּלְמָקוֹם קַר.
  5. מִיהוּ הַיּוֹנֵק שֶׁחַי בַּמַּיִם הַמֻּזְכָּר בַּקֶּטַע?
  6. מַה מְּיֻחָד בָּעֲטַלֵּף?
  7. יָעֵל טָעֲנָה שֶׁתַּרְנְגוֹל הוּא אֵינוֹ יוֹנֵק. הַאִם הִיא צוֹדֶקֶת? הַסְבִּירוּ אֶת תְּשׁוּבַתְכֶם בְּעֶזְרַת הַמֵּידָע מֵהַקֶּטַע.
  8. עִרְכוּ רְשִׁימָה שֶׁל עֲשָׂרָה יוֹנְקִים שֶׁאַתֶּם מַכִּירִים.
  9. נַסְּחוּ שְׁתֵּי שְׁאֵלוֹת עַל יוֹנְקִים שֶׁאֵין עֲלֵיהֶן תְּשׁוּבָה בַּקֶּטַע.

🐍 דַּף עֲבוֹדָה 3: מָה זֶה זוֹחֵל?

📖 קְטַע מֵידָע

תַּנִּינִים, נְחָשִׁים, לְטָאוֹת וְצַבִּים הֵם זוֹחֲלִים. כָּל בַּעֲלֵי הַחַיִּים הָאֵלֶּה הֵם בַּעֲלֵי "דָּם קַר". כְּלוֹמַר, הֵם אֵינָם מְסֻגָּלִים לִשְׁמֹר עַל חֹם גּוּף קָבוּעַ כְּשֶׁמֶּזֶג הָאֲוִיר קַר. הֵם זְקוּקִים לְחֹם הַשֶּׁמֶשׁ, וְלָכֵן רֹב הַזּוֹחֲלִים חַיִּים בָּאֲזוֹרִים חַמִּים. אֵלֶּה הַשּׁוֹכְנִים בִּמְקוֹמוֹת קָרִים, יְשֵׁנִים כָּל הַחֹרֶף וּמִתְעוֹרְרִים בָּאָבִיב. שֵׁנָה זוֹ נִקְרֵאת "תַּרְדֵּמַת חֹרֶף".

💡 פִּנַּת אוֹצַר מִילִּים

בַּעֲלֵי דָּם קַר: חַיּוֹת שֶׁטֶּמְפֶּרָטוּרַת הַגּוּף שֶׁלָּהֶן מִשְׁתַּנָּה בְּהתְאָם לַסְּבִיבָה.

תַּרְדֵּמַת חֹרֶף: שֵׁנָה עֲמֻקָּה וַאֲרֻכָּה בְּמֶשֶׁךְ כָּל עוֹנַת הַחֹרֶף.

📝 הָשִׁיבוּ עַל הַשְּׁאֵלוֹת בְּמַחְבֶּרֶт שָׂפָה

🐠 דַּף עֲבוֹדָה 4: מָה זֶה דָּג?

📖 קְטַע מֵידָע

דָּג מְבַלֶּה אֶת כָּל חַיָּיו בַּמַּיִם. חֵלֶק מֵהַדָּגִים שׁוֹכְנִים בַּיָּם, וְחֵלֶק בִּנְהָרוֹת וּבָאֲגַמִּים. גּוּפָם שֶׁל הַדָּגִים מְכֻסֶּה בְּדֶרֶךְ כְּלָל בְּקַשְׂקַשִּׂים. הַדָּגִים, כְּמוֹ כָּל בַּעֲלֵי הַחַיִּים, זְקוּקִים לְחַמְצָן כְּדֵי לִחְיוֹת. אָנוּ מְקַבְּלִים אֶת הַחַמְצָן מֵהָאֲוִיר, אֲבָל הַדָּגִים מְקַבְּלִים אוֹתוֹ מֵהַמַּיִם. הַמַּיִם נִכְנָסִים אֶל פִּי הַדָּג וְעוֹבְרִים דֶּרֶךְ הַזִּימִים שֶׁלּוֹ. הַזִּימִים קוֹלְטִים מִתּוֹךְ הַמַּיִם אֶת הַחַמְצָן.

💡 פִּנַּת אוֹצַר מִילִּים

זִימִים: אֵיבַר הַנְּשִׁימָה שֶׁל הַדָּגִים, בְּעֶזְרָתוֹ הֵם נוֹשְׁמִים מִתַּחַת לַמַּיִם.

🤔 שְׁאֵלוֹת לְמַחְשָׁבָה

  3. לְמָה זְקוּקִים כָּל בַּעֲלֵי הַחַיִּים כְּדֵי לִחְיוֹת?
  4. מֵאַיִן אָנוּ, בְּנֵי הָאָדָם, מְקַבְּלִים חַמְצָן?
  5. הַסְבִּירוּ בְּמִילִּים שֶׁלָּכֶם, כֵּיצַד הַדָּגִים מְקַבְּלִים חַמְצָן מֵהַמַּיִם?

🐞 דַּף עֲבוֹדָה 5: מָה זֶה חֶרֶק?

📖 קְטַע מֵידָע

לַחֲרָקִים יֵשׁ שֵׁשׁ רַגְלַיִם. גּוּפָם מְכֻסֶּה קְרוּם קָשֶׁה, שֶׁהוּא שֶׁלֶד חִיצוֹנִי. גּוּפוֹ שֶׁל כָּל חֶרֶק מֻרְכָּב מִשְּׁלוֹשָׁה חֲלָקִים: רֹאשׁ, חָזֶה וּבֶטֶן. רֹב הַחֲרָקִים יְכוֹלִים לְעוֹפֵף. לְמַרְבִּיתָם יֵשׁ שְׁנֵי זוּגוֹת כְּנָפַיִם, אֲבָל לַזְּבוּבִים יֵשׁ רַק זוּג כְּנָפַיִם אֶחָד. חֲרָקִים רַבִּים מַרְהִיבִים בְּצִבְעֵיהֶם. צִבְעֵיהֶם הַבּוֹלְטִים מַזְהִירִים אֶת בַּעֲלֵי הַחַיִּים לְבַל יִטְרְפוּ אוֹתָם, כִּי בְּשָׂרָם אֵינוֹ עָרֵב לַחֵךְ.

  4. לְמָה מְשַׁמְּשִׁים הַצְּבָעִים הַבּוֹלְטִים שֶׁיֵּשׁ לַחֲרָקִים?
  5. הַסְבִּירוּ אֶת הַמִּשְׁפָּט: "בְּשָׂרָם אֵינוֹ עָרֵב לַחֵךְ".

🐦 דַּף עֲבוֹדָה 6: מָה זֹאת צִפּוֹר?

📖 קִטְעֵי מֵידָע

1. מָה זֹאת צִפּוֹר?
צִפּוֹר הִיא בַּעַל חַיִּים, שֶׁיֵּשׁ לוֹ נוֹצוֹת וּכְנָפַיִם. צִפֳּרִים חַיּוֹת בְּרֹב אֲזוֹרֵי הָעוֹלָם, וְכִמְעַט כֻּלָּן מְסֻגָּלוֹת לָעוּף. הַצִּפּוֹר הַקְּטַנָּה בְּיוֹתֵר הִיא קוֹלִיבְּרִי זָעִיר, שֶׁאָרְכּוֹ רַק חֲמִשָּׁה סֶנְטִימֶטְרִים. הָעוֹף הַגָּדוֹל בְּיוֹתֵר הוּא הַיָּעֵן, שֶׁגָּבְהוֹ מַגִּיעַ לִשְׁנֵי מֶטְרִים וָחֵצִי.

2. הַנּוֹצוֹת
צִפֳּרִים הֵן בַּעֲלֵי הַחַיִּים הַיְּחִידִים שֶׁיֵּשׁ לָהֶם נוֹצוֹת. הַנּוֹצוֹת נוֹתְנוֹת לַצִּפֳּרִים אֶת צִבְעֵיהֶן וְשׁוֹמְרוֹת עַל חֹם גּוּפָן. הַצִּפֳּרִים מְנַקּוֹת אֶת הַנּוֹצוֹת בְּמַקּוֹרָן וְסָכוֹת אוֹתָן בְּשֶׁמֶן הַנִּפְרָשׁ מִגּוּפָן.

3. הַמָּקוֹר
לְכָל הַצִּפֳּרִים יֵשׁ מַקּוֹר. לְפִי צוּרַת הַמַּקּוֹר אֶפְשָׁר לָדַעַת מִמָּה הֵן נִזּוֹנוֹת. מַקּוֹרֵי הַדְּרוֹרִים קְצָרִים וְעָבִים כְּדֵי לְפַצֵּחַ זְרָעִים. לָעַיִט יֵשׁ מַקּוֹר חַד וְכָפוּף, הַמֻּתְאָם לִקְרִיעַת בָּשָׂר.

  • הִשְׁתַּמְּשׁוּ בִּצְבָעִים וּבְתַצּוּגָה בְּרוּרָה וּמְעַנְיֶנֶת.
  • הַתּוֹצָר צָרִיךְ לִהְיוֹת אֶסְתֵּטִי וּמְסֻדָּר.
  • הָעִקָּר הוּא לֵהָנוֹת מֵהַיְּצִירָה!
6
שֵׁלָב 6: הַצָּגַת הַתּוֹצָר

🎤 כְּלָלִים לַדּוֹבֵר הַמְּצֻיָּן

  • שְׁמִירָה עַל קֶשֶׁר עַיִן: הִסְתַּכְּלוּ עַל הַחֲבֵרִים בַּכִּיתָּה.
  • דִּבּוּר בָּרוּר: דַּבְּרוּ בְּקוֹל רָם, בָּרוּר וְלֹא מַהֵר מִדַּי.
  • עֲמִידָה יַצִּיבָה: עִמְדוּ זָקוּף וְשַׁדְּרוּ בִּטָּחוֹן.
  • הִתְלַהֲבוּת: הַרְאוּ שֶׁהַנּוֹשֵׂא מְעַנְיֵן אֶתְכֶם!
  • שִׁמּוּשׁ בָּעֲזָרִים: הִשְׁתַּמְּשׁוּ בַּתּוֹצָר שֶׁהֲכַנְתֶּם.

👂 כְּלָלִים לַמַּאֲזִין הַטּוֹב

  • הַקְשָׁבָה: נַקְשִׁיב לַדּוֹבֵר עַד סוֹף דְּבָרָיו.
  • רִכּוּז: נִתְרַכֵּז בְּדִבְרֵי הַדּוֹבֵר וְנִתְעַלֵּם מֵהַפְרָעוֹת.
  • כָּבוֹד: נִמָּנַע מִשְּׁאֵלוֹת וְהֶעָרוֹת שֶׁעֲלוּלוֹת לְהַפְרִיעַ לַדּוֹבֵר.

מָשׁוֹב וְרֶפְלֶקְצְיָה

🤔 בּוֹאוּ נַחְשֹׁב עַל תַּהֲלִיךְ הַלְּמִידָה שֶׁעָבַרְנוּ

עָבַרְתֶּם מַסָּע שָׁלֵם! עַכְשָׁיו הַזְּמַן לַעֲצֹר לְרֶגַע וְלַחְשֹׁב: מָה לָמַדְנוּ? מָה הָיָה טוֹב? וּמָה נִרְצֶה לְשַׁפֵּר בַּפַּעַם הַבָּאָה?

✍️ הַמָּשׁוֹב הָאִישִׁי שֶׁלִּי

בְּחִירַת הַנּוֹשֵׂא: הָיִיתִי מְרֻצֶּה מֵהַבְּחִירָה שֶׁלִּי?
😊😐😟
שְׁאִילַת הַשְּׁאֵלוֹת: הַשְּׁאֵלוֹת שֶׁלִּי הָיוּ בְּרוּרוֹת וּמְעַנְיְנוֹת?
😊😐😟
אִסּוּף הַמֵּידָע: הִצְלַחְתִּי לִמְצֹא מֵידָע מִמְּקוֹרוֹת מְגֻוָּנִים?
😊😐😟
אִרְגּוּן וְסִכּוּם: הִצְלַחְתִּי לְסַכֵּם אֶת הַמֵּידָע בְּבֵרוּר?
😊😐😟
הֲכָנַת הַתּוֹצָר: הַתּוֹצָר שֶׁהֵכַנְתִּי יָצָא בָּרוּר, אֶסְתֵּטִי וּמְעַנְיֵן?
😊😐😟
הַצָּגַת הַתּוֹצָר: יִשַּׂמְתִּי אֶת כְּלָלֵי הַדִּבּוּר בִּפְנֵי קהל?
😊😐😟
מַה עָשִׂיתִי הֲכִי טוֹב בְּתַהֲלִיךְ הַחֵקֶר? (הַחֹזְקוֹת שֶׁלִּי)
עִם אֵילוּ קְשָׁיִים וְאֶתְגָּרִים הִתְמוֹדַדְתִּי?
מַה לָּמַדְתִּי עַל עַצְמִי בְּעִקְבוֹת הַחֵקֶר?
